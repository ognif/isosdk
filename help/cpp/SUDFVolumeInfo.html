TCHAR chVolumeLabel[128]; 
The name of the UDF volume label. 
TCHAR chPreparer[128]; 
The name of the data preparer. 
int32 nVersion; 
The version of the UDF filesystem. Possible values are:
BS_UDF_VERSION_102,
BS_UDF_VERSION_150,
BS_UDF_VERSION_200,
BS_UDF_VERSION_201,
BS_UDF_VERSION_250,
BS_UDF_VERSION_260 
int32 nPartitionType; 
The partition type of the UDF filesystem. Possible values are:
BS_UDF_PARTITION_PHYSICAL,
BS_UDF_PARTITION_VIRTUAL,
BS_UDF_PARTITION_SPARABLE 
int32 nMVDSAddress;

int32 nPVDAddress;

int32 nFSDAddress; 
File system descriptor address. 
int32 nVATAddress;

int32 nMetadataAddress; 
Metadata partition address. 
int32 nSparingAddress;

int32 nFileCount; 
The number of files inside the UDF file system. 
int32 nDirCount; 
The number of directories inside the UDF file system.
