DECLARE_STD_EVENT Macro

#define DECLARE_STD_EVENT(name) \
    public:\
        void SetOn##name    (name##Event EraseDone, void *pUserData);\
        name##Event            GetOn##name    (void **pUserData);\
    protected:\
        void name##Thread        (void *pData);\
        name##Event m_On##name##Event;\
        void*        m_pUserDataOn##name##Event;
